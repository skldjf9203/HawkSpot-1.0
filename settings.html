Database Schema

Run these SQL commands in your Supabase SQL Editor to set up the required tables.

-- Table: shop_submissions
CREATE TABLE shop_submissions (
  id            UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  shop_name     TEXT NOT NULL,
  area          TEXT NOT NULL,
  distributor   TEXT NOT NULL,
  submitted_at  TIMESTAMPTZ DEFAULT NOW(),
  submitted_by  TEXT,
  notes         TEXT
);

-- Table: submission_photos
CREATE TABLE submission_photos (
  id             UUID PRIMARY KEY DEFAULT gen_random_uuid(),
  submission_id  UUID REFERENCES shop_submissions(id) ON DELETE CASCADE,
  public_url     TEXT NOT NULL,
  storage_path   TEXT,
  uploaded_at    TIMESTAMPTZ DEFAULT NOW()
);

-- Enable Row Level Security (optional for public read)
ALTER TABLE shop_submissions ENABLE ROW LEVEL SECURITY;
ALTER TABLE submission_photos ENABLE ROW LEVEL SECURITY;

-- Policy: Allow anon read
CREATE POLICY "anon_read_submissions" ON shop_submissions FOR SELECT USING (true);
CREATE POLICY "anon_read_photos" ON submission_photos FOR SELECT USING (true);
